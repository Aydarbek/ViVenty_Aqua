CREATE TABLE Categories (
Id INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
Name NVARCHAR(16) NOT NULL,
Description NVARCHAR(128) NOT NULL
);



CREATE TABLE Hsuits (
Id INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
Name VARCHAR(32) NOT NULL,
Category INT NOT NULL,
Description VARCHAR (256) NOT NULL,
Size VARCHAR (10),
Price INT NOT NULL,
